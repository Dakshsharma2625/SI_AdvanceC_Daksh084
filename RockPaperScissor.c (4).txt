#include <stdio.h>
#include <stdlib.h>

int main()
{
    int user;
    int computer;

    printf("1. Rock\n");
    printf("2. Paper\n");
    printf("3. Scissors\n");

    printf("Enter your choice: ");
    scanf("%d",&user);

    computer = rand()%3 + 1;

    printf("Computer Choice = %d\n",computer);

    if(user == computer)
    {
        printf("Draw");
    }
    else if((user==1 && computer==3) ||
            (user==2 && computer==1) ||
            (user==3 && computer==2))
    {
        printf("You Win");
    }
    else
    {
        printf("Computer Wins");
    }

    return 0;
}
